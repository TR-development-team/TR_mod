			}

	}

		name =TR_Baltic.1.a

			ideology = radical_socialist

		}

			elections_allowed = no
		}
		set_cosmetic_tag = BLT_commune

			limit = {
				is_owned_by = BLT
				NOT = { state = 191 state = 881 state = 883 }
			}
			set_demilitarized_zone = yes
		}	
	}
}
