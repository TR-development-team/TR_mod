country_event = {
	#水力発電所の開業
	id = TR_Austria.23
	title = TR_Austria.23.t
	desc = TR_Austria.23.desc
	picture = GFX_report_event_AUS_hydroelectric_plant
	is_triggered_only = yes
	option = {
		name = TR_Austria.23.a
		if = {
			limit = {
				any_owned_state = {
					has_state_flag = AUS_hydroelectric_construction_planned
					free_building_slots = {
						building = industrial_complex
						size > 1
						include_locked = yes
					}
				}
			}
			random_owned_controlled_state = {
				limit = {
					has_state_flag = AUS_hydroelectric_construction_planned
				}
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				}
			}
		}
		else = {
			add_offsite_building = {
				type = industrial_complex
				level = 1
			}
		}
	}
}
