country_event = {
	id = TR_Soviet.3
	title = TR_Soviet.3.t
	desc = TR_Soviet.3.d
	picture = GFX_report_event_france_parade
	trigger = {
		tag = SOV
        RUS = { exists = no }
        }


        fire_only_once = yes

	
	option = { 
		name = TR_Soviet.3.a
		trigger = {
			NOT = {
				is_in_faction_with = SIB
			}
		}
		set_capital = {
			state = 219
			remember_old_capital = no
		}
        add_timed_idea = {
			idea = GER_CIVILWAR_MARKS
			days = 365
			} 
        remove_ideas = SOV_CIVIL_WAR
		add_stability = 0.1
		hidden_effect = { 
			set_rule = { can_create_factions = no }
			set_global_flag = Soviet_Win_flag
			dismantle_faction = yes
		    every_state = {
		    	limit = {
			    	is_core_of = RUS
		    	}
			    add_core_of = SOV
		    }
		    every_state = {
			    limit = {
				    is_core_of = RUS
			    }
			    remove_core_of = RUS
		    }
	    }
	}
	option = { 
		name = TR_Soviet.3.b
		trigger = {
				is_in_faction_with = SIB
		}
		set_capital = {
			state = 219
			remember_old_capital = no
		}
                add_timed_idea = {
				idea = GER_CIVILWAR_MARKS
				days = 365
				} 
                remove_ideas = SOV_CIVIL_WAR
                set_global_flag = Soviet_Win_flag
		add_stability = 0.1
                set_rule = { can_create_factions = no }
		dismantle_faction = yes
		annex_country = {
			target = SIB
			transfer_troops = yes
			}
		every_state = {
			limit = {
				is_core_of = SIB
			}
			add_core_of = SOV
		}

	}

}
