country_event = {
	id = TR_China.28
	immediate = { log = "[GetDateText]: [Root.GetName]: event TR_China.28 FROM = [From.GetName]" }
	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = {
				has_country_flag = {
					flag = CHI_CCC
					value > 2
				}
			}
			country_event = TR_China.29
		}
		if = {
			limit = {
				has_country_flag = {
					flag = CHI_MIB
					value > 2
				}
			}
			country_event = TR_China.30
		}
		if = {
			limit = {
				has_country_flag = {
					flag = CHI_MIB
					value < 3
				}
				has_country_flag = {
					flag = CHI_CCC
					value < 3
				}
			}
			country_event = TR_China.31
		}
	}
}

#日本の降伏
country_event = {
	id = TR_China.33
	title = TR_China.33.t
	desc = TR_China.33.d
	picture = GFX_report_event_china_chiang_kai_shek

	is_triggered_only = yes
        fire_only_once = yes


	#中国は今、100年の屈辱を晴らしたのだ
	option = {
		name = TR_China.33.a
		ai_chance = {
			base = 100
		}
		set_global_flag = war_of_resistance_china_win
		every_state = {
			limit = {
				OR = { is_core_of = CHI is_claimed_by = CHI }
				is_owned_by = JAP
			}
			transfer_state_to = CHI
		}	
		if = {
			limit = {
				MAN = { exists = yes has_capitulated = yes }
			}
			CHI = {
			    annex_country = {
				    target = MAN
				    transfer_troops = no
				}
		    }
		}
		if = {
			limit = {
				KOR = { exists = yes has_capitulated = yes }
			}
			CHI = {
			    annex_country = {
				    target = KOR
				    transfer_troops = no
				}
		    }
		}
		if = {
			limit = {
				SHX = { exists = yes has_capitulated = yes }
			}
			CHI = {
			    annex_country = {
				    target = SHX
				    transfer_troops = no
				}
		    }
		}
		if = {
			limit = {
				GUG = { exists = yes has_capitulated = yes }
			}
			CHI = {
			    annex_country = {
				    target = GUG
				    transfer_troops = no
				}
		    }
		}
		if = {
			limit = {
				NAN = { exists = yes has_capitulated = yes }
			}
			CHI = {
			    annex_country = {
				    target = NAN
				    transfer_troops = no
				}
		    }
		}
		if = {
			limit = {
				KOR = { exists = no }
			}
			every_state = {
				limit = {
					OR = { is_core_of = KOR }
					is_owned_by = JAP
				}
				transfer_state_to = KOR
			}	
			CHI = {
			    puppet = KOR
		    }
		}
		every_country = {
				limit = {
					OR = { tag = JAP is_in_faction_with = JAP }
				}
				white_peace_with_ROOT = yes
		}	

	}
	#降伏は認めない。日本を地図から消し去るまでは
	option = {
		name = TR_China.33.b
		ai_chance = {
			base = 0
		}
		add_stability = -0.3
	}	
}
