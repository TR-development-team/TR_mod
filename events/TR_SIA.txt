#ビルマとの講和
country_event = {
	id = TR_Siam.7
	title = TR_Siam.7.t
	desc = TR_Siam.7.d
	picture = GFX_report_event_spr_anarchist_uprising

	is_triggered_only = yes
    fire_only_once = yes

	option = {
		name = TR_Siam.7.a
		ai_chance = {
			base = 40
		}
		every_state = {
			limit = {
				is_owned_by = BRM
				OR = { 	
					is_core_of = SIA
					is_claimed_by = SIA
				}
			}
			transfer_state_to = SIA
		}
		white_peace = BRM
		if = {
			limit = { NOT = { has_country_flag = SIA_BRM_border_dispute_end } }
			set_country_flag = SIA_BRM_border_dispute_end
		}
		BRM = { 
			if = {
				limit = { NOT = { has_country_flag = BRM_SIA_border_dispute_end } }
				set_country_flag = BRM_SIA_border_dispute_end
			}
		}
	}
	option = {
		name = TR_Siam.7.b
		ai_chance = {
			base = 0
		}
		add_war_support = -0.2
	}
}
