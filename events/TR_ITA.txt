                }
	}

		name = TR_italy.1.a
                ai_chance = {
			base = 100
		}

			limit = { is_faction_leader = yes }
			dismantle_faction = yes
		}
                every_country = {
                        limit = {
                              has_war_with = ITA
                        }
                        white_peace_with_ROOT = yes
                }
                give_military_access = FRA
                hidden_effect = { 
		    FRA = { 
			country_event = TR_france.42
                    }
		}
        }
        option = { 
		name = TR_italy.1.b
                ai_chance = {
			base = 0
		}
                add_stability = -0.15		
        }
}
