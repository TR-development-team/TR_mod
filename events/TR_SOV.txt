country_event = {
	id = TR_Soviet.3
	title = TR_Soviet.3.t
	desc = TR_Soviet.3.d
	picture = GFX_report_event_france_parade
	trigger = {
		tag = RUS
        SOV = { exists = no }
        }


        fire_only_once = yes

	
	option = { 
		name = TR_Soviet.3.a
		trigger = {
			NOT = {
				is_in_faction_with = SIB
			}
		}
                add_timed_idea = {
				idea = GER_CIVILWAR_MARKS
				days = 365
				} 
                remove_ideas = RUS_CIVIL_WAR
                set_rule = { can_create_factions = no }
                set_global_flag = Soviet_Win_flag
		add_stability = 0.1
		dismantle_faction = yes
	}
	option = { 
		name = TR_Soviet.3.b
		trigger = {
				is_in_faction_with = SIB
		}
                add_timed_idea = {
				idea = GER_CIVILWAR_MARKS
				days = 365
				} 
                remove_ideas = RUS_CIVIL_WAR
                set_global_flag = Soviet_Win_flag
		add_stability = 0.1
                set_rule = { can_create_factions = no }
		dismantle_faction = yes
		annex_country = {
			target = SIB
			transfer_troops = yes
			}
		every_state = {
			limit = {
				is_core_of = SIB
			}
			add_core_of = RUS
		}

	}

}
