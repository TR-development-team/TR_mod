country_event = {
	id = TR_Gansu.1
	title = TR_Gansu.1.t
	desc = TR_Gansu.1.d
	picture = GFX_report_event_generic_rally


    fire_only_once = yes
    trigger = {
        tag = GAN
		SIK = { exists = no }
        ETR = { exists = no }
		KUM = { is_subject_of = GAN }
	}
	

    option = { 
		name = TR_Gansu.1.a
		every_state = {
			limit = {
				is_core_of = SIK
			}
			add_core_of = GAN
		}
		annex_country = {
			target = KUM
			transfer_troops = yes
		}
		set_cosmetic_tag = SIK_ma_military_clique
		set_capital = {
			state = 617
			remember_old_capital = no
		}
	}
}
