#シャムとの講和
country_event = {
	id = TR_Burma.13
	title = TR_Burma.13.t
	desc = TR_Burma.13.d
	picture = GFX_report_event_spr_anarchist_uprising

	is_triggered_only = yes
    fire_only_once = yes

	option = {
		name = TR_Burma.13.a
		ai_chance = {
			base = 40
		}
		every_state = {
			limit = {
				is_owned_by = SIA
				OR = { 	
					is_core_of = BRM
					is_claimed_by = BRM
				}
			}
			transfer_state_to = BRM
		}
		white_peace = SIA
		if = {
			limit = { NOT = { has_country_flag = BRM_SIA_border_dispute_end } }
			set_country_flag = BRM_SIA_border_dispute_end
		}
		SIA = { 
			if = {
				limit = { NOT = { has_country_flag = SIA_BRM_border_dispute_end } }
				set_country_flag = SIA_BRM_border_dispute_end
			}
		}
	}
	option = {
		name = TR_Burma.13.b
		ai_chance = {
			base = 0
		}
		add_war_support = -0.2
	}
}
