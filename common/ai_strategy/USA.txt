

USA_home_land_is_safe = {
	allowed = { original_tag = USA }
	enable = {
		NOT = {
			OR = {
				has_civil_war = yes
				has_war_with = CAN
				has_war_with = MEX
                                is_neighbor_of = CAL
			}
		}
	}
	abort_when_not_enabled = yes
	
	
	ai_strategy = {
		type = garrison
		value = -60
	}
}
