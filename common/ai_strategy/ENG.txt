ENR_protect_Germany = {
	allowed = {
		tag = ENR
	}
	enable = {
		has_war_with = FRA
		is_in_faction_with = GER
		num_divisions > 20
		NOT = {
			surrender_progress > 0
			GER = { has_capitulated = yes }
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = support
		id = "GER"
		value = 200
	}
	ai_strategy = {
		type = send_volunteers_desire
		id = "GER"
		value = 10000
	}
}
