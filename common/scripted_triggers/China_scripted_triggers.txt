is_ruled_by_right_kmt = {
	custom_trigger_tooltip = {
		tooltip = is_ruled_by_right_kmt
		### factions override all alignments
		if = {
			limit = {
				is_in_faction = yes
				is_faction_leader = no
			}
			faction_leader = { original_tag = WHN }
		}
		### overlords override puppets' alignments
		else_if = {
			limit = { is_subject = yes }
			overlord = { original_tag = WHN }
		}
		### if not in faction, check tag-specific conditions
		else = {
			OR = {
				AND = {
					original_tag = GXC
					has_government = national_populist
				}
				AND = {
					original_tag = SCH
					has_government = paternal_autocrat
				}
				AND = {
					original_tag = SIK
					has_government = paternal_autocrat
				}
				AND = {
					original_tag = QIG
					has_government = paternal_autocrat
				}
			}
		}
	}
}
