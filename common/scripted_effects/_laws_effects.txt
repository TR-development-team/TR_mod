increase_trade_laws = {
	if = {
		limit = {
			has_idea = export_focus
		}
		swap_ideas = {
			remove_idea = export_focus
			add_idea = free_trade
		}
	}
	else_if = {
		limit = {
			has_idea = limited_exports
		}
		swap_ideas = {
			remove_idea = limited_exports
			add_idea = export_focus
		}
	}
	else_if = {
		limit = {
			has_idea = closed_economy
		}
		swap_ideas = {
			remove_idea = closed_economy
			add_idea = limited_exports
		}
	}
}
