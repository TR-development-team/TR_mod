			limit = { has_state_category = ten }

		}
		else_if = {
			limit = { has_state_category = eleven }
			set_state_category = eleven_indus
		}
		else_if = {
			limit = { has_state_category = twelve }
			set_state_category = twelve_indus

		limit = {
			OR = {
				has_state_category = five

	}
